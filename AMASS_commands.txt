# AMASS运动数据使用命令示例

## 基础命令格式
python scripts/smplx_to_robot.py --smplx_file <smplx数据路径> --robot <机器人名称> --save_path <保存路径> --rate_limit

## 示例命令（替换为你需要的路径和机器人）

# 使用Male1一般动作数据
python scripts/smplx_to_robot.py \
  --smplx_file /home/z/motion_ws/data/AMASS/ACCAD-G/ACCAD/Male1General_c3d/General_A1_-_Stand_stageii.npz \
  --robot unitree_g1 \
  --save_path ./robot_motion_outputs/stand_motion.pkl \
  --rate_limit

# 使用武术踢腿动作
python scripts/smplx_to_robot.py \
  --smplx_file /home/z/motion_ws/data/AMASS/ACCAD-G/ACCAD/Male2MartialArtsKicks_c3d/G10-__roundhouse_leading_left_stageii.npz \
  --robot unitree_g1 \
  --save_path ./robot_motion_outputs/kick_motion.pkl \
  --rate_limit

# 使用左跳动作
python scripts/smplx_to_robot.py \
  --smplx_file /home/z/motion_ws/data/AMASS/ACCAD-G/ACCAD/Male1Running_c3d/Run_C24_-_quick_side_step_left_stageii.npz \
  --robot unitree_g1 \
  --save_path ./robot_motion_outputs/left_side_jump_motion.pkl \

# 使用右跳动作
